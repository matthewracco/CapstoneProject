4 Replace Auth Middleware

backend/src/middleware/clerkTenant.js
What to change

Replace the stubbed auth with real JWT verification. Keep the BYPASS_AUTH escape hatch for development.

const jwt = require("jsonwebtoken");
const AppError = require("../utils/AppError");

const JWT_SECRET = process.env.JWT_SECRET || "smartlocker-secret-key-change-in-prod";

function requireAuth(req, res, next) {
  // Keep bypass for dev
  if (process.env.BYPASS_AUTH === "true") {
    req.user = {
      userId: process.env.DEV_USER_ID || "user_demo",
      role: "OWNER",
      tenantId: process.env.DEV_TENANT_ID || "tenant_demo",
    };
    return next();
  }

  // Real auth: extract Bearer token from Authorization header
  const authHeader = req.headers.authorization;
  if (!authHeader?.startsWith("Bearer ")) {
    return next(new AppError("No token provided", 401, "UNAUTHORIZED"));
  }

  const token = authHeader.split(" ")[1];
  try {
    const decoded = jwt.verify(token, JWT_SECRET);
    req.user = decoded; // { userId, tenantId, role }
    next();
  } catch (err) {
    next(new AppError("Invalid or expired token", 401, "TOKEN_INVALID"));
  }
}

function requireTenant(req, res, next) {
  // If auth passed, tenantId is already in req.user from JWT
  if (process.env.BYPASS_AUTH === "true") {
    req.tenantId = process.env.DEV_TENANT_ID || "tenant_demo";
    return next();
  }

  req.tenantId = req.user?.tenantId;
  if (!req.tenantId) {
    return next(new AppError("Tenant required", 400, "TENANT_REQUIRED"));
  }
  next();
}

module.exports = { requireAuth, requireTenant };

5 RBAC Middleware (Role Check)

backend/src/middleware/rbac.js
What this file does + code

Simple middleware that checks if the logged-in user has the required role.

const AppError = require("../utils/AppError");

// Usage: router.get("/admin-only", requireAuth, requireTenant, requireRole("OWNER", "STAFF"), handler)
function requireRole(...allowedRoles) {
  return (req, res, next) => {
    if (process.env.BYPASS_AUTH === "true") return next();

    const userRole = req.user?.role;
    if (!userRole || !allowedRoles.includes(userRole)) {
      return next(new AppError("Insufficient permissions", 403, "FORBIDDEN"));
    }
    next();
  };
}

module.exports = { requireRole };

import { createContext, useState, useEffect } from "react";

export const AuthContext = createContext();

export function AuthProvider({ children }) {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // On mount, check localStorage for existing auth
    const stored = localStorage.getItem("auth");
    if (stored) {
      const parsed = JSON.parse(stored);
      setUser(parsed.user);
    }
    setLoading(false);
  }, []);

  const login = (data) => {
    // data = { user, token, refreshToken } from API response
    localStorage.setItem("auth", JSON.stringify(data));
    setUser(data.user);
  };

  const logout = () => {
    localStorage.removeItem("auth");
    setUser(null);
  };

  return (
    <AuthContext.Provider value={{ user, login, logout, loading }}>
      {children}
    </AuthContext.Provider>
  );
}

10 Protected Route Component

frontend/src/app/components/ProtectedRoute.jsx
What this file does

Wraps routes that require login. If no user, redirect to /login.

import { Navigate, Outlet } from "react-router-dom";
import { useAuth } from "../../features/auth/useAuth";

export default function ProtectedRoute({ allowedRoles }) {
  const { user, loading } = useAuth();

  if (loading) return <div>Loading...</div>;
  if (!user) return <Navigate to="/login" />;
  if (allowedRoles && !allowedRoles.includes(user.role)) {
    return <Navigate to="/" />;
  }
  return <Outlet />;
}

Feature 4: Access Code System

When a user rents a locker, they generate a 6-digit code to "unlock" it. Code expires after 5 minutes.

flowchart TD
    USER[User taps 'Get Access Code'] --> API["POST /api/v1/rentals/:id/access-code"]
    API --> CHECK{Rental is ACTIVE\nand belongs to user?}
    CHECK -->|No| ERR[Return 403 error]
    CHECK -->|Yes| RATE{Generated < 3 codes\nin last 15 min?}
    RATE -->|No| RATE_ERR[Return 429 Too Many Requests]
    RATE -->|Yes| GEN["Generate random 6-digit code\nMath.floor(100000 + Math.random() * 900000)"]
    GEN --> HASH["Hash code with bcrypt\nbcrypt.hash(code, 10)"]
    HASH --> STORE["Store in Locker:\naccessCode = hashedCode\naccessCodeExpiry = now + 5min"]
    STORE --> RETURN["Return plain code to user\n(only time it's shown)"]
    RETURN --> DISPLAY["Frontend shows:\n\n  8 4 7 2 9 1\n  Expires in 4:59"]

    style ERR fill:#f85149,color:#fff
    style RATE_ERR fill:#f85149,color:#fff
    style DISPLAY fill:#3fb950,color:#000

async function generateAccessCode({ tenantId, userId, rentalId }) {
  // 1. Find rental, verify it's ACTIVE and belongs to userId

Cron Jobs (Background Tasks)

backend/src/jobs/expiryChecker.js
Code structure 
const cron = require("node-cron");
const prisma = require("../config/prisma");
const notificationService = require("../services/notification.service");

function startExpiryChecker() {
  // Every 5 minutes: check for rentals expiring within 1 hour
  cron.schedule("*/5 * * * *", async () => {
    const soon = new Date(Date.now() + 60 * 60 * 1000); // 1 hour from now
    const expiring = await prisma.rental.findMany({
      where: { status: "ACTIVE", endTime: { lte: soon, gt: new Date() } },
      include: { locker: true },
    });
    for (const rental of expiring) {
      await notificationService.create({
        tenantId: rental.tenantId,
        userId: rental.userId,
        type: "EXPIRY_WARNING",
        title: "Rental Expiring Soon",
        body: `Your rental for locker ${rental.locker.lockerNumber} expires soon.`,
      });
    }
  });

  // Every minute: check for expired rentals, auto-complete them
  cron.schedule("* * * * *", async () => {
    const expired = await prisma.rental.findMany({
      where: { status: "ACTIVE", endTime: { lte: new Date() } },
    });
    for (const rental of expired) {
      // Force complete the rental and free up the locker
      await prisma.$transaction([
        prisma.rental.update({ where: { id: rental.id }, data: { status: "COMPLETED" } }),
        prisma.locker.updateMany({ where: { id: rental.lockerId }, data: { status: "AVAILABLE" } }),
      ]);
      await notificationService.create({
        tenantId: rental.tenantId, userId: rental.userId,
        type: "RENTAL_EXPIRED", title: "Rental Expired",
        body: "Your rental has expired and the locker has been freed.",
      });
    }
  });
}

module.exports = { startExpiryChecker };
